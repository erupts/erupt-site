{{'cube.ml.fn.reg.title' | i18n}}
@CubeFunction("fn")
public class SqlFunctions {
    // usage: ${fn.inClause(ids)} → 'a', 'b', 'c'
    public String inClause(List<String> ids) {
        return ids.stream()
            .map(id -> "'" + id + "'")
            .collect(Collectors.joining(", "));
    }
}

◈ Erupt Cube (Java Annotations)
@EruptCube(
  name = "Order Analysis",
  sql = "orders",
  sqlType = SqlType.TABLE_NAME
)
public class OrderCube {

  @Dimension(title = "Channel")
  private String channel;

  @Dimension(title = "Status")
  private String status;

  @Measure(
    title = "Orders",
    sql = "count(*)"
  )
  private Long orderCount;

  @Measure(
    title = "Revenue",
    sql = "sum(amount)"
  )
  private Double totalAmount;

}
