OrderCube.java 
@EruptCube(
    name = "订单分析",
    sql = """
        select o.channel, o.status, o.create_time,
               o.amount, u.city, p.category
        from orders o
          inner join users u on o.user_id = u.id
          inner join products p on o.product_id = p.id
        """
)
public class OrderCube {

    // ── 维度：用于分组与筛选 ──────────────────────────
    @Dimension(title = "渠道", sql = "channel")
    private String channel;

    @Dimension(title = "状态", sql = "status")
    private String status;

    @Dimension(title = "城市", sql = "city")
    private String city;

    @Dimension(title = "品类", sql = "category")
    private String category;

    @Dimension(title = "下单时间", sql = "create_time")
    private Date createTime;

    // ── 指标：聚合计算，一次定义到处复用 ─────────────
    @Measure(title = "订单量", sql = "count(*)")
    private Long orderCount;

    @Measure(title = "销售额", sql = "sum(amount)")
    private Double totalAmount;

    @Measure(title = "客单价", sql = "round(sum(amount) / count(*), 2)")
    private Double avgOrderValue;

    @Measure(title = "独立用户数", sql = "count(distinct user_id)")
    private Long uniqueUsers;
}

◈ Erupt Cube (Java Annotations) 
@EruptCube(
  name = "订单分析",
  sql = "orders",
  sqlType = SqlType.TABLE_NAME
)
public class OrderCube {

  @Dimension(title = "渠道")
  private String channel;

  @Dimension(title = "状态")
  private String status;

  @Measure(
    title = "订单量",
    sql = "count(*)")
  private Long orderCount;

  @Measure(
    title = "销售额",
    sql = "sum(amount)")
  private Double totalAmount;
}
