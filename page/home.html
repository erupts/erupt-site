/**
 * plug-and-play 🚀  1 class replaces:
 * html / js / css / controller / service / dao / mapper / ddl

@Table
@Entity
@Erupt(
    name = "任务管理",
    power = @Power(export = true, importable = true),
    filter = @Filter("dept_id = #{user.deptId}"),
    dataProxy = TaskProxy.class,
    rowOperation = @RowOperation(title = "重置状态"),
    vis = @Vis(title = "看板", type = Vis.Type.CARD, cardView = @CardView)
)
public class Task extends BaseModel {

    @EruptField(
        views = @View(title = "优先级"),
        edit  = @Edit(title = "优先级", type = EditType.CHOICE,
                      choiceType = @ChoiceType(vl = {
                          @VL(label = "高", value = "high", color = "#ef4444"),
                          @VL(label = "中", value = "mid",  color = "#f59e0b"),
                          @VL(label = "低", value = "low",  color = "#22c55e")
                      }))
    )
    private String priority;

    @EruptField(
        views = @View(title = "位置", type = ViewType.MAP),
        edit  = @Edit(title = "位置", type = EditType.MAP)
    )
    private String location;

    @EruptField(
        edit = @Edit(title = "脚本", type = EditType.CODE_EDITOR,
                     codeEditType = @CodeEditorType(language = "sql"))
    )
    private String script;

    @EruptField(
        views = @View(title = "审批签名", type = ViewType.IMAGE_BASE64),
        edit  = @Edit(title = "审批签名", type = EditType.SIGNATURE)
    )
    private String signature;

    @OneToMany(cascade = CascadeType.ALL, orphanRemoval = true)
    @JoinColumn(name = "task_id")
    @EruptField(
        edit = @Edit(title = "子任务", type = EditType.TAB_TABLE_ADD)
    )
    private Set<SubTask> children;

}
