/**
 * plug-and-play 🚀  1 class replaces:
 * html / js / css / controller / service / dao / mapper / ddl
 */
@Table
@Entity
@Erupt(
    name = "Task Management",
    power = @Power(export = true),
    dataProxy = TaskProxy.class,
    rowOperation = @RowOperation(title = "Reset Status"),
    vis = @Vis(title = "Board", type = Vis.Type.CARD)
)
public class Task extends BaseModel {

    @EruptField(
        views = @View(title = "Priority"),
        edit  = @Edit(title = "Priority", type = EditType.CHOICE,
                      choiceType = @ChoiceType(vl = {
                          @VL(label = "High",   value = "high", color = "#ef4444"),
                          @VL(label = "Medium", value = "mid",  color = "#3b82f6"),
                          @VL(label = "Low",    value = "low",  color = "#3b82f6")
                      }))
    )
    private String priority;

    @EruptField(
        views = @View(title = "Location", type = ViewType.MAP),
        edit  = @Edit(title = "Location", type = EditType.MAP)
    )
    private String location;

    @EruptField(
        edit = @Edit(title = "Script", type = EditType.CODE_EDITOR,
                     codeEditType = @CodeEditorType(language = "sql"))
    )
    private String script;

    @EruptField(
        views = @View(title = "Approval Sign", type = ViewType.IMAGE_BASE64),
        edit  = @Edit(title = "Approval Sign", type = EditType.SIGNATURE)
    )
    private String signature;

    @OneToMany(cascade = CascadeType.ALL, orphanRemoval = true)
    @JoinColumn(name = "task_id")
    @EruptField(
        edit = @Edit(title = "Sub Tasks", type = EditType.TAB_TABLE_ADD)
    )
    private Set<SubTask> children;

}
